public class ChallengeV {
    public static void addPhone() {
        List<Contact> noPhoneC = [SELECT phone, account.phone FROM Contact WHERE Phone = null];
        for (Contact c : noPhoneC) {
            c.Phone = c.account.phone;
        }
        System.debug(noPhoneC);
        update noPhoneC;
    }
   
    public static void TakingItToTask() {
        List<Task> taskList = new List<Task>();
        List<Contact> noEmailC = [SELECT Email, Name, OwnerId FROM Contact WHERE Email = null];
        for(Contact c : noEmailC) {
            Task bobsTask = new Task();
            bobsTask.OwnerId = c.OwnerId;
            bobsTask.Subject = 'Update Email';
            taskList.add(bobsTask);
            System.debug(taskList);
        }
        Insert taskList;
    }
}
